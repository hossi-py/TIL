## 1) S5. 1205. 등수구하기

```python
# S5. 1205. 등수구하기

# N : 리스트에 있는 점수의 개수 / 송유진 점수 / 올라 갈 수 있는 개수
N, Song, P = map(int, input().split())

if N == 0:
    print(1)

else:
    scores = list(map(int, input().split()))

    # 랭킹 리스트가 꽉 차있고, Song의 점수가 제일 낮을 때
    if N == P and scores[-1] >= Song:
        print(-1)

    else:
        res = N + 1
        for i in range(N):
            if scores[i] <= Song:
                res = i + 1
                break
        print(res)

```



## 2) S5. 1251. 단어나누기

```python
word = input()
new_words = []  # 만들어진 단어들

for cut in range(1, len(word) - 1):
    f_word = word[:cut]     # 잘린 맨 앞 단어
    cut_word = word[cut:]   # 한 번 잘리고 남은 단어
    rf_word = f_word[::-1]  # 맨 앞 단어 뒤집기

    for plus in range(1, len(cut_word)):
        # 남은 단어 잘릴 수 있는 경우의 수
        m_word = cut_word[:plus]
        l_word = cut_word[plus:]

        # 단어 뒤집기
        rm_word = m_word[::-1]
        rl_word = l_word[::-1]

        # 문자 만들고 추가하기
        new_words.append(rf_word + rm_word + rl_word)

new_words.sort()

print(new_words[0])

```



## 3) S5. 1292. 쉽게 푸는 문제

```python
A, B = map(int ,input().split())

arr = []

cnt = 1

while len(arr) <= 1000:
    for _ in range(cnt):
        arr.append(cnt)

    cnt += 1

result = 0

for num in range(A - 1, B):
    result += arr[num]

print(result)

```
