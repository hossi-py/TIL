        x, y = q.popleft()
        if graph[x][y] == -1:

## 3. [S3 / 3182 / 한동이는 공부가 하기 싫어!]

```python
# S3. 3182. 한동이는 공부가 하기 싫어!

N = int(input())    # 선배들 수


G = {i + 1: int(input()) for i in range(N)}     # key : 선배 번호 value : 대답

counts = [0] * (N + 1)  # 각 선배를 시작으로 몇 번 만나는지 저장
path = [0] * (N + 1)    # 방문 배열


def recursive(cur):
    if path[cur] == 1:  # cycle 발생 (이미 만났을 때)
        return 0

    path[cur] = 1   # 방문 체크

    # 실행 2 이후 cnt를 늘려 만난 선배 수를 늘린다
    cnt = recursive(G[cur]) + 1 # 다음 선배

    # 1번 선배를 먼저 만났을 때 다시 1번 선배를 만나게 된다면 시작되는 코드
    # 실행 1
    path[cur] = 0
    # 실행 2
    return cnt

for i in range(1, N + 1):
    counts[i] = recursive(i)

print(counts.index(max(counts)))
```



## 4. [S3 / 14562 / 태권왕]

```python
from collections import deque

# S3. 14562. 태권왕

# A : 현재 점수만큼 점수를 얻을 수 있는 기술 / 상대 역시 3점을 득점
# B : 1점을 얻는 기술

# S : 태균이의 현재 점수 / T : 상대의 점수
# S == T가 되는 최소 연속 발차기 횟수
def bfs(S, T):
    q = deque()
    q.append((S, T, 0))

    while q:
        t, e, cnt = q.popleft()
        # 태균이 점수가 낮을 때
        if t <= e:
            # 기술 A 썼을 때
            q.append((t * 2, e + 3, cnt + 1))
            # 기술 B 썼을 때
            q.append((t + 1, e, cnt + 1))

            # 점수 같으면
            if t == e:
                return cnt


C = int(input())

for _ in range(C):
    S, T = map(int, input().split())

    print(bfs(S, T))

```

        # 늑대라면 이동

                if 0 <= nx < R and 0 <= ny < C:

dx = [-1, 0, 1, 0]

def dfs(x, y):
    global wolf, sheep

    if farm[x][y] == 'v':
        wolf += 1
        farm[x][y] = '#'

    if farm[x][y] == 'k':
        sheep += 1
        farm[x][y] = '#'

    for k in range(4):
        nx = x + dx[k]
        ny = y + dy[k]

        if 0 <= nx < R and 0 <= ny < C:
            # 늑대일 때
            if farm[nx][ny] == 'v':
                # 늑대 마리수 1 증가
                wolf += 1
                farm[nx][ny] = '#'
                dfs(nx, ny)

            # 양일 때
            elif farm[nx][ny] == 'k':
                # 양 마리수 1 증가
                sheep += 1
                farm[nx][ny] = '#'
                dfs(nx, ny)

            elif farm[nx][ny] == '.':
                farm[nx][ny] = '#'
                dfs(nx, ny)

farm = [list(input()) for _ in range(R)]

total_wolf, total_sheep = 0, 0

for i in range(R):
    for j in range(C):

            # 울타리 안에 있는 늑대, 양 마리 수
            wolf, sheep = 0, 0

            # 울타리마다 비교
            # 늑대가 양보다 많거나 같을 때
            if wolf >= sheep:
                sheep = 0
            # 양이 더 많을 때
            elif wolf < sheep:
                wolf = 0

            # 잡아 먹힌 결과
            total_wolf += wolf
            total_sheep += sheep

print(total_sheep, total_wolf)

## 8. [S3 / 17204 / 죽음의 게임]

```python
from collections import deque
# S3. 17204. 죽음의 게임
def playGame(cur, d):
    next = gameList[cur]    # 다음 사람

    # 지목 당한 사람 또 당하면 반복
    if checkList[cur] == 1:
        return 0

    # 게임 진행
    checkList[cur] = 1  # cur idx 사람 체크
    depthList[cur] = d  # 현재사람 지목 횟수

    playGame(next, d + 1)


N, K = map(int, input().split())

gameList = [int(input()) for _ in range(N)] # 지목당하는 사람
checkList = [0] * N     # 지목 여부 확인
depthList = [-1] * N    # 게임 계속 될 때마다 언제 지목 당했는지 확인

playGame(0, 0)
print(depthList[K]) # 보성이가 몇번 째에 처음 지목되는지

```
