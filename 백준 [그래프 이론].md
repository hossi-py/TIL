```python
# 14397. 해변
dx = [-1, -1, 0, 0, 1, 1]
# x의 인덱스가 0포함 짝수일 때
dy_even = [-1, 0, -1, 1, -1, 0]
# 홀수일 때
dy_odd = [0, 1, -1, 1, 0, 1]


def dfs(x, y):
    global beach_len

    for k in range(6):
        nx = x + dx[k]

        if x % 2 == 0:
            ny = y + dy_even[k]
        else:
            ny = y + dy_odd[k]

        if 0 <= nx < N and 0 <= ny < M:
            if beach[nx][ny] == '#':
                beach_len += 1

# 세로의 크기 N / 가로의 크기 M
N, M = map(int, input().split())

beach = [list(input()) for _ in range(N)]

beach_len = 0
# 물의 위치에서 8방향으로 땅과 만나면 체크
for i in range(N):
    for j in range(M):
        if beach[i][j] == '.':
            dfs(i, j)

print(beach_len)

```
