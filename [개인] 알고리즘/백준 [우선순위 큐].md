## 16466. 콘서트

```python
'''
1. 이미 들어 있는거 확인할 때는 "in" 보다 visited (False, True)가 빠르다.
2. 중복된 경우 확인 and 티켓번호가 모두 1일 때 2가 출력 되지 않았음
'''

# 16466. 콘서트
import heapq

N = int(input())

info = list(map(int, input().split()))

visited = [False] * 1000001

q = []

for i in range(len(info)):
    if visited[info[i]] is False:
        heapq.heappush(q, info[i])
        visited[info[i]] = True

idx = 1

check = False

while q:
    ticket_num = heapq.heappop(q)

    if ticket_num == idx:
        idx += 1

    else:
        print(idx)
        check = True
        break

if not check:
    print(idx)

```



## 1417. 국회의원 선거

```python
'''
매 순간(q에서 뽑을 때 마다) 최댓값을 비교한다. => 우선순위 큐
'''
# 1417. 국회의원 선거
import sys, heapq

N = int(input())

dasom = int(input())

q = []

for _ in range(N - 1):
    candi = int(input())

    heapq.heappush(q, -candi)   # 최대 힙

buy = 0

while q:
    candi = -heapq.heappop(q)

    if dasom > candi:
        break

    buy += 1    # 매수
    dasom += 1  # 표 획득
    candi -= 1  # 다른 후보 표 제거

    heapq.heappush(q, -candi)

print(buy)

```



## 12018. Yonsei TOTO

```python
'''
# 조건 0에서 잔여 마일리지 확인을 안함
'''
import sys, heapq
# 12018. Yonsei TOT
n, mile = map(int, input().split())
ans = 0
q = []

for i in range(n):
    p, l = map(int, sys.stdin.readline().split())

    if p < l:   # 조건 0
        info = list(map(int, input().split()))
        if mile > 0:
            ans += 1
            mile -= 1

    else:
        info = list(map(int, input().split()))
        info.sort(reverse=True)
        heapq.heappush(q, info[l - 1])

while q:
    use = heapq.heappop(q)

    if mile >= use:
        mile -= use
        ans += 1

    else:
        break

print(ans)

```



## 14235. 크리스마스 선물

```python
import sys, heapq
# 14235. 크리스마스 선물
n = int(input())

q = []

for i in range(n):
    info = list(map(int, sys.stdin.readline().split()))

    if len(info) > 1:
        for i in range(1, len(info)):
            heapq.heappush(q, -info[i])

    if info[0] == 0:    # 아이들을 만났을 때
        if len(q) > 0:    # 줄 선물이 있다면
            give = -heapq.heappop(q)    # 가치가 가장 큰 선물 주기
            print(give)

        else:
            print(-1)
```



## 11279. 최대 힙

```python
import sys, heapq
# 11279. 최대 힙
n = int(input())

q = []

for _ in range(n):
    x = int(sys.stdin.readline().rstrip())

    if x == 0:  # 배열에서 가장 큰 값을 출력하고 그 값을 배열에서 제거
        if len(q) == 0: # 배열이 비어있는 경우
            print(0)

        else:
            num = -heapq.heappop(q)
            print(num)

    else:
        heapq.heappush(q, -x)

```



## 15903. 카드 합체 놀이

```python
import sys, heapq
# 15903. 카드 합체 놀이
n, m = map(int, input().split())

cards = list(map(int, input().split()))

_sum = sum(cards)

q = []

for i in range(n):
    heapq.heappush(q, cards[i])

if m == 0:
    print(_sum)

else:
    while q:
        if m == 0:
            break

        temp = 0    # 갱신을 위한 값

        # x, y 뽑기
        x = heapq.heappop(q)
        y = heapq.heappop(q)

        temp += (x + y) # 섞음
        m -= 1

        for _ in range(2):
            heapq.heappush(q, temp)

    print(sum(q))

```
