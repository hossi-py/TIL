## 16466. 콘서트

```python
'''
1. 이미 들어 있는거 확인할 때는 "in" 보다 visited (False, True)가 빠르다.
2. 중복된 경우 확인 and 티켓번호가 모두 1일 때 2가 출력 되지 않았음
'''

# 16466. 콘서트
import heapq

N = int(input())

info = list(map(int, input().split()))

visited = [False] * 1000001

q = []

for i in range(len(info)):
    if visited[info[i]] is False:
        heapq.heappush(q, info[i])
        visited[info[i]] = True

idx = 1

check = False

while q:
    ticket_num = heapq.heappop(q)

    if ticket_num == idx:
        idx += 1

    else:
        print(idx)
        check = True
        break

if not check:
    print(idx)

```



## 1417. 국회의원 선거

```python
'''
매 순간(q에서 뽑을 때 마다) 최댓값을 비교한다. => 우선순위 큐
'''
# 1417. 국회의원 선거
import sys, heapq

N = int(input())

dasom = int(input())

q = []

for _ in range(N - 1):
    candi = int(input())

    heapq.heappush(q, -candi)   # 최대 힙

buy = 0

while q:
    candi = -heapq.heappop(q)

    if dasom > candi:
        break

    buy += 1    # 매수
    dasom += 1  # 표 획득
    candi -= 1  # 다른 후보 표 제거

    heapq.heappush(q, -candi)

print(buy)

```



## 12018. Yonsei TOTO

```python
'''
# 조건 0에서 잔여 마일리지 확인을 안함
'''
import sys, heapq
# 12018. Yonsei TOT
n, mile = map(int, input().split())
ans = 0
q = []

for i in range(n):
    p, l = map(int, sys.stdin.readline().split())

    if p < l:   # 조건 0
        info = list(map(int, input().split()))
        if mile > 0:
            ans += 1
            mile -= 1

    else:
        info = list(map(int, input().split()))
        info.sort(reverse=True)
        heapq.heappush(q, info[l - 1])

while q:
    use = heapq.heappop(q)

    if mile >= use:
        mile -= use
        ans += 1

    else:
        break

print(ans)

```



## 14235. 크리스마스 선물

```python
import sys, heapq
# 14235. 크리스마스 선물
n = int(input())

q = []

for i in range(n):
    info = list(map(int, sys.stdin.readline().split()))

    if len(info) > 1:
        for i in range(1, len(info)):
            heapq.heappush(q, -info[i])

    if info[0] == 0:    # 아이들을 만났을 때
        if len(q) > 0:    # 줄 선물이 있다면
            give = -heapq.heappop(q)    # 가치가 가장 큰 선물 주기
            print(give)

        else:
            print(-1)
```

## 15903. 카드 합체 놀이

```python
import sys, heapq
# 15903. 카드 합체 놀이
n, m = map(int, input().split())

cards = list(map(int, input().split()))

_sum = sum(cards)

q = []

for i in range(n):
    heapq.heappush(q, cards[i])

if m == 0:
    print(_sum)

else:
    while q:
        if m == 0:
            break

        temp = 0    # 갱신을 위한 값

        # x, y 뽑기
        x = heapq.heappop(q)
        y = heapq.heappop(q)

        temp += (x + y) # 섞음
        m -= 1

        for _ in range(2):
            heapq.heappush(q, temp)

    print(sum(q))

```



## 17503. 맥주축제

```python
'''
sum 함수의 Big-O 는 O(N) 이다. => 시간 초과 조심
'''
import sys, heapq
# 17503. 맥주 축제
N, M, K = map(int, input().split())

info, q = [], []

for _ in range(K):
    v, c = map(int, sys.stdin.readline().split())
    info.append((v, c))

# 도수 낮은 순 정렬
info.sort(key=lambda x: x[1])

now, _sum = 0, 0

flag = False

for favor, degree in info:
    heapq.heappush(q, favor)
    now = degree
    _sum += favor

    if len(q) == N:
        if _sum >= M:
            flag = True
            break

        else:
            _sum -= heapq.heappop(q)

if flag:
    print(now)
else:
    print(-1)

```



## 23757. 아이들과 선물 상자

```python
import sys, heapq
# 23757. 아이들과 선물 상자
N, M = map(int, input().split())
have = list(map(int, input().split()))
want = list(map(int, input().split()))

q = []

for h in have:
    heapq.heappush(q, -h)

flag = True

for w in want:
    remain_gifts = -heapq.heappop(q)

    if remain_gifts >= w:   # 선물을 아이에게 줄 수 있음
        heapq.heappush(q, -(remain_gifts - w))# 주기

    else:
        flag = False

if flag:
    print(1)
else:
    print(0)

```

## 19638. 센티와 마법의 뿅망치

```python
'''
반례가 많은 문제
'''
import sys, heapq
N, H, T = map(int, input().split())

q = []

for _ in range(N):
    heapq.heappush(q, -int(sys.stdin.readline().rstrip()))

cnt = 0 # 최소 뿅망치 사용 횟수

flag = True

while T > 0:
    big = -heapq.heappop(q) # 가장 큰 거인

    if big == 1 and big == H:    # 가장 큰 거인의 키가 1이고 센티도 1일 때
        flag = False
        heapq.heappush(q, -big)
        break

    if big >= H:    # 거인의 키가 센티보다 크면
        heapq.heappush(q, -(big // 2))  # 마법
        T -= 1  # 뿅망치 사용
        cnt += 1

    else:   # 뿅망치 모두 사용 전에 다 작아짐
        heapq.heappush(q, -big)
        break

if T > 0 and flag is True:
    print("YES")
    print(cnt)

else:
    big = -heapq.heappop(q)

    if big >= H:
        print("NO")
        print(big)

    else:
        print("YES")
        print(cnt)

```

## 11000. 강의실 배정

```python
import sys, heapq
# 11000. 강의실 배정
N = int(input())

task = []

for _ in range(N):
    t = list(map(int, sys.stdin.readline().split()))

    task.append(t)

# 시작 시간 순 정렬
task.sort(key=lambda x: x[0])
q = []
heapq.heappush(q, task[0][1])   # q배열에 첫번째 강의의 종료 시간을 넣는다.

for i in range(1, N):
    if q[0] > task[i][0]:   # 종료시간 > 다른 강의의 시작 시간
        heapq.heappush(q, task[i][1])   # 연달아 강의할 수 없으므로 다른 강의의 종료 시간을 q에 넣는다

    else:    # 종료시간 <= 다른 강의의 시작 시간
        heapq.heappop(q)    # 같은 강의실에서 강의 가능하므로 이전 강의 정보 제거
        heapq.heappush(q, task[i][1])   # 새로운 강의의 종료시간 넣기

print(len(q))

```



## 1374. 강의실

```python
'''
11000. 강의실 배정과 같은 문제
'''
import sys, heapq
# 1374. 강의실
N = int(input())

info = []

for _ in range(N):
    n, s, e = map(int, sys.stdin.readline().split())
    info.append([s, e])

# 시작 시간 순 정렬
info.sort(key=lambda x: x[0])
q = []
heapq.heappush(q, info[0][1])  # 시작 시간이 가장 이른 강의의 종료 시간을 배열에 넣는다.

for i in range(1, N):
    # 종료 시간이 다른 강의의 시작 시간 보다 큰 경우 : 연달아 불가능
    if q[0] > info[i][0]:
        heapq.heappush(q, info[i][1])   # 강의실 추가 후 다음 강의 비교

    else:
        heapq.heappop(q)    # 첫번째 배열 강의 제거
        heapq.heappush(q, info[i][1])   # 연강하는 강의의 종료시간 입력

print(len(q))

```
